128 int xemusock_init ( char *msg )
129 {
130  *msg = '\0';
131  if (_winsock_init_status <= 0)
132  return _winsock_init_status;
133 #ifdef XEMU_ARCH_WIN
134  WSADATA wsa;
135  if (WSAStartup(MAKEWORD(WINSOCK_VERSION_MAJOR, WINSOCK_VERSION_MINOR), &wsa)) {
136  if (msg) {
137  int err = SOCK_ERR();
138  sprintf(msg, "WINSOCK: ERROR: Failed to initialize winsock2, [%d]: %s", err, xemusock_strerror(err));
139  }
140  _winsock_init_status = -1;
141  return -1;
142  }
143  if (LOBYTE(wsa.wVersion) != WINSOCK_VERSION_MAJOR || HIBYTE(wsa.wVersion) != WINSOCK_VERSION_MINOR) {
144  WSACleanup();
145  if (msg)
146  sprintf(msg,
147  "WINSOCK: ERROR: No suitable winsock API in the implemantion DLL (we need v%d.%d, we got: v%d.%d), windows system error ...",
148  WINSOCK_VERSION_MAJOR, WINSOCK_VERSION_MINOR,
149  HIBYTE(wsa.wVersion), LOBYTE(wsa.wVersion)
150  );
151  _winsock_init_status = -1;
152  return -1;
153  }
154  if (msg)
155  sprintf(msg, "WINSOCK: OK: initialized, version %d.%d", HIBYTE(wsa.wVersion), LOBYTE(wsa.wVersion));
156 #endif
157  _winsock_init_status = 0;
158  return 0;
159 }

183 int xemusock_set_nonblocking ( xemusock_socket_t sock, int is_nonblock, int *xerrno )
184 {
185 #ifdef XEMU_ARCH_WIN
186  u_long mode = !!is_nonblock; // 1 = use non-blocking sockets
187  if (ioctlsocket(sock, FIONBIO, &mode)) {
188  if (xerrno)
189  *xerrno = SOCK_ERR();
190  return -1;
191  }
192  return 0;
193 #else
194  int flags = fcntl(sock, F_GETFL);
195  if (flags == -1) {
196  if (xerrno)
197  *xerrno = SOCK_ERR();
198  return -1;
199  }
200  flags = (is_nonblock ? (flags | O_NONBLOCK) : (flags & ~O_NONBLOCK));
201  if (fcntl(sock, F_SETFL, flags) == -1) {
202  if (xerrno)
203  *xerrno = SOCK_ERR();
204  return -1;
205  }
206  return 0;
207 #endif
208 }

214  if (sock == XS_INVALID_SOCKET) {

221  if (ret) {

224  } else

226  }

318 int xemusock_select_1 ( xemusock_socket_t sock, int usec, int what )
319 {
320  for (;;) {
321  int ret;
322  struct timeval timeout;
323  fd_set fds_r, fds_w, fds_e;
324  FD_ZERO(&fds_r);
325  FD_ZERO(&fds_w);
326  FD_ZERO(&fds_e);
327  if (what & XEMUSOCK_SELECT_R)
328  FD_SET(sock, &fds_r);
329  if (what & XEMUSOCK_SELECT_W)
330  FD_SET(sock, &fds_w);
331  if (what & XEMUSOCK_SELECT_E)
332  FD_SET(sock, &fds_e);
333  timeout.tv_sec = 0;
334  timeout.tv_usec = usec;
335  ret = select(sock + 1, &fds_r, &fds_w, &fds_e, usec >= 0 ? &timeout : NULL);
336  if (ret == XS_SOCKET_ERROR) {
337  int err = SOCK_ERR();
338  if (err == XSEINTR)
339  continue;
340  return -1;
341  }
342  if (ret == 0)
343  return 0;
344  return (FD_ISSET(sock, &fds_r) ? XEMUSOCK_SELECT_R : 0) | (FD_ISSET(sock, &fds_w) ? XEMUSOCK_SELECT_W : 0) | (FD_ISSET(sock, &fds_e) ? XEMUSOCK_SELECT_E : 0);
345  }
346 }
